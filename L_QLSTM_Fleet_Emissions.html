L-QLSTM Fleet Emissions Forecaster¶

Based on: Linear-Layer-Enhanced QLSTM (Cao et al.)¶

Paper Relevance¶

Paper (Carbon Price) Our Problem (Fleet Emissions)
EU ETS time-series Fleet diesel/energy time-series
5 OHLCV features 20 features (4 streams x 5 machines)
1 output (price) 5 outputs (CO2e per machine)
QLSTM fails: 56% qubit waste Same failure: 83% waste (20 feat, 4 qubits)
L-QLSTM fixes with shared Le Same fix applied here
SCZ: all-to-all entanglement Captures fuel/elec/heat correlations

Three Paper Innovations¶

  1. Shared Le before VQCs (Eq.9) — compress to N_QUBITS, measure 100% of qubits
  2. Separate Li after each VQC (Section 3.2) — gate-specific remapping
  3. SCZ variational layer (Section 3.3, Fig.8) — CRZ all-to-all vs CNOT chain

Forward Pass (Equations 9-15)¶

v_t = [h_{t-1}, x_t]
z_t = Le(v_t)                         # Eq.9  shared linear embedding
f_t = sigmoid( L1( VQC1(z_t) ) )      # Eq.10 forget gate
i_t = sigmoid( L2( VQC2(z_t) ) )      # Eq.11 input gate
g_t = tanh( L3( VQC3(z_t) ) )         # Eq.12 cell gate
c_t = f_t * c_{t-1} + i_t * g_t      # Eq.13 cell state
o_t = sigmoid( L4( VQC4(z_t) ) )      # Eq.14 output gate
h_t = o_t * tanh(c_t)                 # Eq.15 hidden state
y_t = Lo(h_t)                          # Eq.16 output

def generate_data(machines, n_days=365, start='2024-01-01'):
    rng = np.random.default_rng(42)
    dates = pd.date_range(start, periods=n_days, freq='D')
    records = []
    for mname, p in machines.items():
        maint = set(range(0, n_days, 45))
        for i, ts in enumerate(dates):
            wkend  = 0.35 if ts.dayofweek >= 5 else 1.0
            season = 1.0 + 0.12 * np.sin(2 * np.pi * ts.dayofyear / 365)
            mf     = 0.15 if i in maint else 1.0
            load   = rng.uniform(0.65, 1.0) * wkend * season * mf
            fuel   = max(0, p['fuel']*24*load + rng.normal(0, p['fuel']*2))
            elec   = max(0, p['elec']*24*load + rng.normal(0, p['elec']*1.2))
            heat   = max(0, p['heat']*24*load + rng.normal(0, p['heat']*1.0))
            refrig = p['ra'] if rng.random() < p['rp']*24 else 0.0
            co2e   = fuel*EF['fuel'] + refrig*EF['refrigerant'] + elec*EF['electricity'] + heat*EF['heat']
            records.append(dict(date=ts, machine=mname,
                fuel_L=round(fuel,2), refrigerant_kg=round(refrig,4),
                electricity_kWh=round(elec,2), heat_MJ=round(heat,2), co2e_kg=round(co2e,2)))
    return pd.DataFrame(records)

Section 4 — Quantum Device and SCZ Circuit¶

SCZ (Fig. 8 of paper) vs CNOT (Fig. 7):

  • CNOT: chain connectivity, rotation AFTER entanglement
  • CRZ all-to-all: every pair connected, rotation DURING entanglement
  • Paper result: 34.83% RMSE reduction with SCZ over CNOT

N_QUBITS = 4   # = hidden_size: ALL qubits measured, 0% waste
N_LAYERS = 2   # SCZ variational depth

# lightning.qubit: PennyLane's optimised C++ simulator — ~10x faster than
# qiskit.aer for small circuits.  Already installed via pennylane-lightning.
# diff_method='adjoint': computes gradients in a single forward+backward pass
# (O(params) cost) instead of parameter-shift's 2×params circuit evaluations.
# To switch to real IBM hardware later, swap only this device line — the rest
# of the model is unchanged.
dev = qml.device('lightning.qubit', wires=N_QUBITS)
print(f"Device: {dev.name}")

def scz_layer(weights, n_qubits):
    """
    SCZ layer (Fig. 8 of paper):
    1. RZ on every qubit  (pre-entanglement rotation)
    2. CRZ between ALL pairs  (all-to-all entanglement with rotation)
    3. RZ on every qubit  (post-entanglement rotation)
    """
    for i in range(n_qubits):
        qml.RZ(weights[i], wires=i)
    for i in range(n_qubits):
        for j in range(i+1, n_qubits):
            qml.CRZ(weights[i], wires=[i,j])
    for i in range(n_qubits):
        qml.RZ(weights[n_qubits+i], wires=i)

@qml.qnode(dev, interface='torch', diff_method='adjoint')
def vqc_scz(inputs, weights):
    """
    VQC with SCZ — replaces one LSTM gate.
    inputs : (N_QUBITS,) from Le embedding
    weights: (N_LAYERS, N_QUBITS*2)
    returns: N_QUBITS Pauli-Z values in [-1,1]
    ALL qubits measured (Le ensures all carry information)
    """
    qml.AngleEmbedding(inputs, wires=range(N_QUBITS), rotation='Y')
    for l in range(N_LAYERS):
        scz_layer(weights[l], N_QUBITS)
    return [qml.expval(qml.PauliZ(w)) for w in range(N_QUBITS)]

class LQLSTMCell(nn.Module):
    """
    L-QLSTM Cell from Cao et al.

    Key design (Section 3.1 & 3.2):
    - Shared Le: ONE linear layer for all 4 VQCs (parameter sharing, cross-gate dependency)
    - Separate Li: FOUR linear layers after VQCs (each gate has different function)
    - SCZ VQC: all-to-all entanglement (Section 3.3)
    """
    def __init__(self, input_size, hidden_size):
        super().__init__()
        self.hidden_size = hidden_size
        combined = input_size + hidden_size

        # Eq(9): Le — shared, maps to N_QUBITS
        self.Le = nn.Linear(combined, N_QUBITS)

        # VQC trainable weights — one per gate
        vqc_shape = (N_LAYERS, N_QUBITS * 2)
        self.vqc_weights = nn.ParameterDict({
            'forget': nn.Parameter(torch.randn(vqc_shape) * 0.1),
            'input' : nn.Parameter(torch.randn(vqc_shape) * 0.1),
            'cell'  : nn.Parameter(torch.randn(vqc_shape) * 0.1),
            'output': nn.Parameter(torch.randn(vqc_shape) * 0.1),
        })

        # Eq(10-14): Li — separate per gate
        self.L1 = nn.Linear(N_QUBITS, hidden_size)  # forget
        self.L2 = nn.Linear(N_QUBITS, hidden_size)  # input
        self.L3 = nn.Linear(N_QUBITS, hidden_size)  # cell
        self.L4 = nn.Linear(N_QUBITS, hidden_size)  # output

    def _vqc(self, z, gate):
        w = self.vqc_weights[gate]
        # .float() casts float64 (qiskit.aer expectation values) → float32 (nn.Linear weights)
        return torch.stack([torch.stack(vqc_scz(z[i], w)) for i in range(z.shape[0])]).float()

    def forward(self, x_t, h_prev, c_prev):
        v_t = torch.cat([h_prev, x_t], dim=-1)      # concatenate
        z_t = torch.tanh(self.Le(v_t))              # Eq(9): shared Le
        q_f = self._vqc(z_t, 'forget')
        q_i = self._vqc(z_t, 'input')
        q_g = self._vqc(z_t, 'cell')
        q_o = self._vqc(z_t, 'output')
        f_t = torch.sigmoid(self.L1((q_f+1)/2))     # Eq(10)
        i_t = torch.sigmoid(self.L2((q_i+1)/2))     # Eq(11)
        g_t = torch.tanh   (self.L3(q_g))           # Eq(12)
        o_t = torch.sigmoid(self.L4((q_o+1)/2))     # Eq(14)
        c_t = f_t * c_prev + i_t * g_t             # Eq(13)
        h_t = o_t * torch.tanh(c_t)                # Eq(15)
        return h_t, c_t

class BiLQLSTM(nn.Module):
    """Bidirectional L-QLSTM — extension of paper's unidirectional model."""
    def __init__(self, input_size, hidden_size):
        super().__init__()
        self.hs  = hidden_size
        self.fwd = LQLSTMCell(input_size, hidden_size)
        self.bwd = LQLSTMCell(input_size, hidden_size)

    def forward(self, x):
        B = x.size(0)
        hf=cf=torch.zeros(B,self.hs)
        hb=cb=torch.zeros(B,self.hs)
        for t in range(x.size(1)):
            hf,cf = self.fwd(x[:,t,:],hf,cf)
        for t in reversed(range(x.size(1))):
            hb,cb = self.bwd(x[:,t,:],hb,cb)
        return torch.cat([hf,hb],dim=-1)

class FleetLQLSTM(nn.Module):
    """Fleet model: BiLQLSTM + 5 Lo heads (Eq.16 per machine)."""
    def __init__(self, input_size, hidden_size, n_machines):
        super().__init__()
        self.encoder = BiLQLSTM(input_size, hidden_size)
        self.heads = nn.ModuleList([
            nn.Sequential(nn.Linear(2*hidden_size,8),nn.ReLU(),nn.Linear(8,1),nn.Sigmoid())
            for _ in range(n_machines)
        ])
    def forward(self, x):
        s = self.encoder(x)
        return torch.cat([h(s) for h in self.heads], dim=-1)

def autoregressive_forecast(model, X_sc, scaler_y, lb, n):
    model.eval(); w=X_sc[-lb:].copy(); out=[]
    with torch.no_grad():
        for _ in range(n):
            p=model(torch.tensor(w[np.newaxis],dtype=torch.float32)).squeeze(0).numpy()
            out.append(p); w=np.vstack([w[1:],w[-1]])
    return scaler_y.inverse_transform(np.array(out))
